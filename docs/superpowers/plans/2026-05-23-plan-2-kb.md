# CorpMind — Plan 2: Documents + Knowledge Assistant (RAG)

> **For agentic workers:** Executed via Codex CLI orchestrated by Claude. Tasks use checkbox (`- [ ]`). Spec: `docs/superpowers/specs/2026-05-23-corpmind-mvp-design.md` (§3.4 hybrid RAG, §3.5 chunking, §3.6 KB prompt, §3.12 SSE, §4.1 endpoints). Prerequisite: Plan 1 (Foundation) merged.

**Goal:** Make Knowledge Assistant work end-to-end — admin uploads PDF/DOCX, system ingests/chunks/embeds, employee asks a question in natural language, gets a grounded streamed answer with source citations and a contextual next-step suggestion.

**Architecture:** A local-filesystem-backed `storage` module abstracts file IO. Document upload returns immediately and enqueues a BullMQ `documents.ingest` job that extracts text (pdf-parse/mammoth), chunks (~500 tokens, 80 overlap), batch-embeds (OpenAI text-embedding-3-small), and inserts rows + pgvector embeddings via raw SQL. RAG service runs both pgvector cosine and tsvector keyword searches, fuses via Reciprocal Rank Fusion (k=60), takes top-5, and builds the KB-answer prompt. NestJS `@Sse()` streams gpt-4o-mini tokens back; on completion the trailing citations JSON is parsed and persisted as `Citation[]`, then a `next-step` suggestion is generated in parallel. Web side: a ChatStream component renders streamed tokens with citation chips and a "what to do next" callout.

**Tech Stack additions (over Plan 1):** `openai` SDK, `pdf-parse`, `mammoth`, `tiktoken`, BullMQ workers, `@bull-board/express`, `recharts` (precursor — used in Plan 4).

**Definition of Done:**
- Admin uploads 6 seed docs via `pnpm ingest:demo-docs`; all show `status=READY` within ~2 minutes
- `POST /api/v1/kb/ask` (SSE) streams a grounded answer to a Uzbek question with ≥1 citation
- Web `/kb` page shows streaming tokens + clickable citation chips that open the source document
- "Next step" suggestion appears under each answer (or "no further steps" if none)
- Conversation history persists; messages can be rated thumbs up/down
- `pnpm typecheck && pnpm lint && pnpm test` green; new Playwright spec `kb-ask.spec.ts` green

```typescript
import { Injectable, Logger, NotFoundException } from '@nestjs/common';
import * as fs from 'node:fs/promises';
import * as path from 'node:path';
import { createReadStream } from 'node:fs';
import { randomUUID } from 'node:crypto';
import { loadEnv } from '../config/env.js';

@Injectable()
export class StorageService {
  private readonly logger = new Logger(StorageService.name);
  private readonly root: string;
  private readonly publicBase: string;

  constructor() {
    const env = loadEnv();
    this.root = path.resolve(env.STORAGE_LOCAL_DIR);
    this.publicBase = env.STORAGE_PUBLIC_URL;
  }

  async ensureRoot(): Promise<void> {
    await fs.mkdir(this.root, { recursive: true });
  }

  async putBuffer(tenantId: string, originalName: string, data: Buffer): Promise<{ key: string; size: number }> {
    await this.ensureRoot();
    const safeName = originalName.replace(/[^a-zA-Z0-9._-]/g, '_').slice(-120);
    const key = path.posix.join('tenant', tenantId, `${Date.now()}-${randomUUID().slice(0, 8)}-${safeName}`);
    const abs = this.absolute(key);
    await fs.mkdir(path.dirname(abs), { recursive: true });
    await fs.writeFile(abs, data);
    return { key, size: data.byteLength };
  }

  absolute(key: string): string {
    const abs = path.resolve(this.root, key);
    if (!abs.startsWith(this.root + path.sep) && abs !== this.root) {
      throw new Error('Path escape detected');
    }
    return abs;
  }

  publicUrl(key: string): string {
    return `${this.publicBase}/${key}`;
  }

  async readBuffer(key: string): Promise<Buffer> {
    return fs.readFile(this.absolute(key));
  }

  streamFile(key: string): NodeJS.ReadableStream {
    return createReadStream(this.absolute(key));
  }

  async exists(key: string): Promise<boolean> {
    try { await fs.stat(this.absolute(key)); return true; } catch { return false; }
  }

  async remove(key: string): Promise<void> {
    try { await fs.unlink(this.absolute(key)); } catch (e) {
      this.logger.warn(`remove failed for ${key}: ${(e as Error).message}`);
    }
  }
}
```

```typescript
import { Queue, Worker, type ConnectionOptions } from 'bullmq';
import IORedis from 'ioredis';
import { loadEnv } from '../config/env.js';
import { QUEUE_NAMES, type QueueName } from './queues.js';

let connection: ConnectionOptions | undefined;

export function getConnection(): ConnectionOptions {
  if (!connection) {
    const env = loadEnv();
    connection = new IORedis(env.REDIS_URL, { maxRetriesPerRequest: null });
  }
  return connection;
}

const queues = new Map<QueueName, Queue>();
export function getQueue(name: QueueName): Queue {
  let q = queues.get(name);
  if (!q) {
    q = new Queue(name, { connection: getConnection(), defaultJobOptions: { attempts: 3, backoff: { type: 'exponential', delay: 5000 } } });
    queues.set(name, q);
  }
  return q;
}

export type WorkerProcessor<T> = (data: T, jobId: string) => Promise<void>;
export function startWorker<T>(name: QueueName, processor: WorkerProcessor<T>): Worker<T> {
  return new Worker<T>(name, async (job) => processor(job.data as T, job.id ?? 'unknown'), { connection: getConnection(), concurrency: 2 });
}

export const QUEUES = QUEUE_NAMES;
```

```typescript
import { Injectable, Logger } from '@nestjs/common';
import IORedis from 'ioredis';
import { loadEnv } from '../config/env.js';

type TelemetryEntry = {
  model: string;
  promptTokens: number;
  completionTokens: number;
  latencyMs: number;
};

// Approx USD per 1M tokens (input, output). Update as needed.
const PRICING: Record<string, { in: number; out: number }> = {
  'gpt-4o':                  { in: 2.5,  out: 10.0 },
  'gpt-4o-mini':             { in: 0.15, out: 0.6 },
  'text-embedding-3-small':  { in: 0.02, out: 0 },
};

@Injectable()
export class TelemetryService {
  private readonly redis: IORedis;
  private readonly logger = new Logger(TelemetryService.name);
  private readonly dailyBudgetUsd: number;

  constructor() {
    const env = loadEnv();
    this.redis = new IORedis(env.REDIS_URL, { lazyConnect: true, maxRetriesPerRequest: 1 });
    this.dailyBudgetUsd = env.OPENAI_DAILY_BUDGET_USD;
  }

  async record(entry: TelemetryEntry): Promise<{ costUsd: number; dailyUsd: number }> {
    const p = PRICING[entry.model] ?? { in: 0, out: 0 };
    const cost = (entry.promptTokens * p.in + entry.completionTokens * p.out) / 1_000_000;
    const dayKey = `ai:cost:${new Date().toISOString().slice(0, 10)}`;
    if (this.redis.status !== 'ready') await this.redis.connect().catch(() => {});
    const total = await this.redis.incrbyfloat(dayKey, cost);
    await this.redis.expire(dayKey, 60 * 60 * 36);
    const dailyUsd = Number(total);
    this.logger.log(`${entry.model} ${entry.promptTokens}p+${entry.completionTokens}c ${entry.latencyMs}ms $${cost.toFixed(4)} (day $${dailyUsd.toFixed(2)})`);
    if (dailyUsd > this.dailyBudgetUsd) {
      throw new Error(`OpenAI daily budget exceeded ($${dailyUsd.toFixed(2)} > $${this.dailyBudgetUsd})`);
    }
    return { costUsd: cost, dailyUsd };
  }
}
```

- [ ] **Step 4: `apps/api/src/ai/embeddings/chunker.ts`** — boundary-aware token chunker

```typescript
import { encoding_for_model, type Tiktoken } from 'tiktoken';

const CHUNK_TARGET_TOKENS = 500;
const CHUNK_OVERLAP_TOKENS = 80;

export type Chunk = {
  text: string;
  tokenCount: number;
  headerHint: string | null;
};

let encoderSingleton: Tiktoken | null = null;
function encoder(): Tiktoken {
  if (!encoderSingleton) encoderSingleton = encoding_for_model('text-embedding-3-small');
  return encoderSingleton;
}

function splitParagraphs(input: string): string[] {
  return input.split(/\n{2,}/).map((s) => s.trim()).filter(Boolean);
}

function splitSentences(input: string): string[] {
  return input
    .replace(/\s+/g, ' ')
    .split(/(?<=[.!?])\s+(?=[A-ZА-ЯЎҚҒҲ])/u)
    .map((s) => s.trim())
    .filter(Boolean);
}

function detectHeader(paragraph: string): string | null {
  const trimmed = paragraph.trim();
  if (trimmed.length < 120 && /^[#0-9.IVX\-]/.test(trimmed)) return trimmed.slice(0, 120);
  if (trimmed.toUpperCase() === trimmed && trimmed.length < 120) return trimmed;
  return null;
}

export function chunkText(rawText: string): Chunk[] {
  const enc = encoder();
  const chunks: Chunk[] = [];
  let currentHeader: string | null = null;
  let buffer: number[] = [];

  const flush = (): void => {
    if (buffer.length === 0) return;
    const decoded = new TextDecoder().decode(enc.decode(new Uint32Array(buffer)));
    chunks.push({ text: decoded.trim(), tokenCount: buffer.length, headerHint: currentHeader });
    // Apply overlap by retaining the tail
    buffer = buffer.slice(Math.max(0, buffer.length - CHUNK_OVERLAP_TOKENS));
  };

  for (const para of splitParagraphs(rawText)) {
    const maybeHeader = detectHeader(para);
    if (maybeHeader) {
      currentHeader = maybeHeader;
      continue;
    }
    for (const sentence of splitSentences(para)) {
      const tokens = Array.from(enc.encode(sentence + ' '));
      if (buffer.length + tokens.length > CHUNK_TARGET_TOKENS) flush();
      buffer.push(...tokens);
    }
  }
  if (buffer.length > 0) {
    const decoded = new TextDecoder().decode(enc.decode(new Uint32Array(buffer)));
    chunks.push({ text: decoded.trim(), tokenCount: buffer.length, headerHint: currentHeader });
  }
  return chunks;
}
```

- [ ] **Step 6: `apps/api/src/ai/embeddings/embeddings.service.ts`**

```typescript
import { Injectable } from '@nestjs/common';
import IORedis from 'ioredis';
import { createHash } from 'node:crypto';
import { OpenAiClient } from '../openai.client.js';
import { TelemetryService } from '../telemetry.service.js';
import { loadEnv } from '../../config/env.js';

const BATCH = 32;
const CACHE_TTL_SEC = 60 * 60;

@Injectable()
export class EmbeddingsService {
  private readonly redis: IORedis;
  private readonly model: string;
  constructor(private readonly openai: OpenAiClient, private readonly telemetry: TelemetryService) {
    const env = loadEnv();
    this.redis = new IORedis(env.REDIS_URL, { lazyConnect: true, maxRetriesPerRequest: 1 });
    this.model = env.OPENAI_MODEL_EMBED;
  }

  async embedQuery(text: string): Promise<number[]> {
    const key = `embed:q:${this.model}:${createHash('sha1').update(text).digest('hex')}`;
    if (this.redis.status !== 'ready') await this.redis.connect().catch(() => {});
    const cached = await this.redis.get(key);
    if (cached) return JSON.parse(cached) as number[];
    const [vec] = await this.embedBatch([text]);
    if (!vec) throw new Error('empty embedding response');
    await this.redis.set(key, JSON.stringify(vec), 'EX', CACHE_TTL_SEC);
    return vec;
  }

  async embedBatch(texts: string[]): Promise<number[][]> {
    const out: number[][] = [];
    for (let i = 0; i < texts.length; i += BATCH) {
      const slice = texts.slice(i, i + BATCH);
      const t0 = Date.now();
      const res = await this.openai.raw.embeddings.create({ model: this.model, input: slice });
      const latency = Date.now() - t0;
      const totalTokens = res.usage?.total_tokens ?? 0;
      await this.telemetry.record({
        model: this.model, promptTokens: totalTokens, completionTokens: 0, latencyMs: latency,
      });
      for (const item of res.data) out.push(item.embedding as number[]);
    }
    return out;
  }
}
```
